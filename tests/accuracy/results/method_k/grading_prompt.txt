### 用例 7（复杂过滤，难度2）
**问题**：找出客单价在500到2000之间、25-34岁或35-44岁女性用户的已完成订单
**参考 SQL**：
```sql
SELECT dwd_order_detail.order_id, dim_user.user_name, dim_user.age_group, dwd_order_detail.total_amount
FROM dwd_order_detail
INNER JOIN dim_user ON dwd_order_detail.user_id = dim_user.user_id
WHERE dwd_order_detail.order_status = '已完成'
  AND dwd_order_detail.total_amount BETWEEN 500 AND 2000
  AND dim_user.gender = 'female'
  AND dim_user.age_group IN ('25-34', '35-44')
```

**run-1**：
```sql
WITH user_avg_order AS (
  SELECT dwd_order_detail.user_id, AVG(dwd_order_detail.total_amount) AS avg_order_amount
  FROM dwd_order_detail
  INNER JOIN dim_user ON dwd_order_detail.user_id = dim_user.user_id
  WHERE dwd_order_detail.order_status = '已完成' AND dim_user.gender = 'female' AND (dim_user.age_group = '25-34' OR dim_user.age_group = '35-44')

)
SELECT dwd_order_detail.order_id, dwd_order_detail.user_id, dim_user.user_name, dim_user.age_group, dwd_order_detail.order_dt, dwd_order_detail.total_amount, user_avg_order.avg_order_amount

INNER JOIN dim_user ON dwd_order_detail.user_id = dim_user.user_id
INNER JOIN user_avg_order ON dwd_order_detail.user_id = user_avg_order.user_id
WHERE dwd_order_detail.order_status = '已完成' AND dim_user.gender = 'female' AND (dim_user.age_group = '25-34' OR dim_user.age_group = '35-44') AND user_avg_order.avg_order_amount >= 500 AND user_avg_order.avg_order_amount <= 2000
```

**run-2**：

**run-3**：

### 用例 40（综合复杂查询，难度3）
**问题**：各年龄段用户的退款率（退款订单数/总完成订单数），并按退款率排名
**参考 SQL**：
```sql
WITH age_orders AS (
  SELECT dim_user.age_group,
    COUNT(DISTINCT dwd_order_detail.order_id) AS total_orders
  FROM dwd_order_detail
  INNER JOIN dim_user ON dwd_order_detail.user_id = dim_user.user_id
  WHERE dwd_order_detail.order_status = '已完成'
  GROUP BY dim_user.age_group
),
age_refunds AS (
  SELECT dim_user.age_group,
    COUNT(DISTINCT dwd_refund_detail.refund_id) AS refund_count
  FROM dwd_refund_detail
  INNER JOIN dim_user ON dwd_refund_detail.user_id = dim_user.user_id
  GROUP BY dim_user.age_group
)
SELECT age_orders.age_group, age_orders.total_orders,
  COALESCE(age_refunds.refund_count, 0) AS refund_count,
  ROUND(CAST(COALESCE(age_refunds.refund_count, 0) AS REAL) / age_orders.total_orders, 4) AS refund_rate,
  RANK() OVER (ORDER BY ROUND(CAST(COALESCE(age_refunds.refund_count, 0) AS REAL) / age_orders.total_orders, 4) DESC) AS refund_rank
FROM age_orders
LEFT JOIN age_refunds ON age_orders.age_group = age_refunds.age_group
ORDER BY refund_rank
```

**run-1**：
```sql
WITH age_group_stats AS (
  SELECT dim_user.age_group, COUNT(*) AS total_orders, COUNT(dwd_refund_detail.refund_id) AS refund_orders
  FROM dwd_order_detail
  INNER JOIN dim_user ON dwd_order_detail.user_id = dim_user.user_id
  LEFT JOIN dwd_refund_detail ON dwd_order_detail.order_id = dwd_refund_detail.order_id
  WHERE dwd_order_detail.order_status = '已完成'
  GROUP BY dim_user.age_group
)
SELECT age_group, total_orders, refund_orders, CASE WHEN total_orders > 0 THEN CAST(refund_orders AS FLOAT) / total_orders ELSE 0 END AS refund_rate
FROM age_group_stats
ORDER BY refund_rate DESC
```
**run-2**：
```sql
WITH user_order_stats AS (
  SELECT dim_user.user_id, dim_user.age_group, COUNT(*) AS completed_orders, SUM(CASE WHEN dwd_refund_detail.refund_id IS NOT NULL THEN 1 ELSE 0 END) AS refund_orders
  FROM dim_user
  LEFT JOIN dwd_order_detail ON dim_user.user_id = dwd_order_detail.user_id
  LEFT JOIN dwd_refund_detail ON dwd_order_detail.order_id = dwd_refund_detail.order_id
  WHERE dwd_order_detail.order_status = '已完成'
  GROUP BY dim_user.user_id, dim_user.age_group
)
SELECT user_order_stats.age_group, SUM(completed_orders) AS total_completed_orders, SUM(refund_orders) AS total_refund_orders, CAST(SUM(refund_orders) AS REAL) / SUM(completed_orders) AS refund_rate
FROM user_order_stats
GROUP BY user_order_stats.age_group
ORDER BY refund_rate DESC
```
**run-3**：
```sql
WITH user_order_refund AS (
  SELECT dim_user.age_group, dwd_order_detail.order_id, CASE WHEN dwd_refund_detail.refund_status IN ('已退款', '审核通过') THEN 1 ELSE 0 END AS is_refunded
  FROM dim_user
  LEFT JOIN dwd_order_detail ON dim_user.user_id = dwd_order_detail.user_id
  LEFT JOIN dwd_refund_detail ON dwd_order_detail.order_id = dwd_refund_detail.order_id
  WHERE dwd_order_detail.order_status = '已完成'

SELECT age_group, COUNT(*) AS total_orders, SUM(is_refunded) AS refund_orders, CAST(SUM(is_refunded) AS FLOAT) / CAST(COUNT(*) AS FLOAT) AS refund_rate
FROM user_order_refund

ORDER BY refund_rate DESC
```
